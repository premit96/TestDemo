#include<iostream>
#include<cstring>
using namespace std;
int main()
{
   string str="";
   int len,flag=0;
   cout<<"enter a string"<<endl;
   getline(cin, str);

   len=strlen(str);
   char *p1,*p2;
   p1=str;
   p2=str+len-1;
   while(p1<p2)
   {
       if(*p1==*p2)
       flag=0;
       else
       flag++;
       p1++;
       p2--;
   }
   if(flag==0)
   cout<<"PALINDROME"<<endl
   else
   cout<<" NOT A PALINDROME"<<endl
   return 0;
}
